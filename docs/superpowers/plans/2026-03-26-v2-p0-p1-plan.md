```typescript
import { v5 as uuidv5 } from "uuid";

/** Custom namespace UUID for UltraMemory (constant, never changes) */
const ULTRAMEMORY_NAMESPACE = "7f3e8a2b-4c5d-4e6f-8a9b-1c2d3e4f5a6b";

/**
 * Generate a deterministic UUID v5 from scope + text.
 * Same scope+text always produces the same ID.
 * NUL separator prevents scope/text boundary ambiguity.
 */
export function deterministicId(scope: string, text: string): string {
  const input = `${scope}\0${text}`;
  return uuidv5(input, ULTRAMEMORY_NAMESPACE);
}
```

- [ ] **Step 1: Implement `fuseMultiChannelResults()`**

Add to `packages/core/src/retriever.ts`:

```typescript
interface ChannelResult {
  results: MemorySearchResult[];
  weight: number;
}

function fuseMultiChannelResults(
  channels: ChannelResult[],
  rrfK: number,
  bm25Results: MemorySearchResult[],
): Map<string, RetrievalResult> {
  const fusionMap = new Map<string, RetrievalResult>();

  for (const channel of channels) {
    for (let rank = 0; rank < channel.results.length; rank++) {
      const r = channel.results[rank];
      const rrfScore = channel.weight / (rrfK + rank + 1);
      const existing = fusionMap.get(r.entry.id);
      if (existing) {
        existing.score += rrfScore;
      } else {
        fusionMap.set(r.entry.id, {
          entry: r.entry,
          score: rrfScore,
          sources: {},
        } as RetrievalResult);
      }
    }
  }

  // BM25 high-score floor: if bm25Score >= 0.75, ensure fused >= bm25Score * 0.92
  for (const bm25r of bm25Results) {
    if (bm25r.score >= 0.75) {
      const existing = fusionMap.get(bm25r.entry.id);
      if (existing && existing.score < bm25r.score * 0.92) {
        existing.score = bm25r.score * 0.92;
      }
    }
  }

  return fusionMap;
}
```

Create `packages/core/src/feedback-learner.ts`:

```typescript
import { parseSmartMetadata, stringifySmartMetadata, buildSmartMetadata } from "./smart-metadata.js";

export class FeedbackLearner {
  constructor(
    private store: { getById(id: string): Promise<any>; patchMetadata(id: string, patch: any): Promise<any> },
    private alpha: number = 0.15,
  ) {}

  async recordPositive(memoryId: string): Promise<void> {
    await this.applyEMA(memoryId, 1.0);
  }

  async recordNegative(memoryId: string): Promise<void> {
    await this.applyEMA(memoryId, 0.0);
  }

  async recordExplicit(memoryId: string, helpful: boolean): Promise<void> {
    await this.applyEMA(memoryId, helpful ? 1.0 : 0.0);
  }

  async setDirect(memoryId: string, weight: number): Promise<void> {
    const entry = await this.store.getById(memoryId);
    if (!entry) return;
    await this.store.patchMetadata(memoryId, { feedback_weight: Math.max(0, Math.min(1, weight)) });
  }

  private async applyEMA(memoryId: string, signal: number): Promise<void> {
    const entry = await this.store.getById(memoryId);
    if (!entry) return;
    const meta = parseSmartMetadata(entry.metadata, entry);
    const oldWeight = meta.feedback_weight ?? 0.5;
    const newWeight = this.alpha * signal + (1 - this.alpha) * oldWeight;
    await this.store.patchMetadata(memoryId, { feedback_weight: Math.max(0, Math.min(1, newWeight)) });
  }
}
```

- [ ] **Step 3: Implement `applyRelationBoost` in retriever.ts**

```typescript
export const RELATION_BOOST_PER_HIT = 0.03;
export const RELATION_BOOST_CAP = 0.09;

export function applyRelationBoost(
  candidates: Array<{ entry: { id: string; metadata?: string }; score: number }>,
): typeof candidates {
  const poolIds = new Set(candidates.map((c) => c.entry.id));

  // Build reverse index: targetId → Set<sourceId>
  const reverseIndex = new Map<string, Set<string>>();
  const forwardMap = new Map<string, string[]>(); // sourceId → targetIds in pool

  for (const c of candidates) {
    const meta = typeof c.entry.metadata === "string" ? JSON.parse(c.entry.metadata) : {};
    const relations: Array<{ targetId: string }> = meta.relations ?? [];
    const forwardHits: string[] = [];
    for (const rel of relations) {
      if (poolIds.has(rel.targetId)) forwardHits.push(rel.targetId);
      const set = reverseIndex.get(rel.targetId);
      if (set) set.add(c.entry.id);
      else reverseIndex.set(rel.targetId, new Set([c.entry.id]));
    }
    forwardMap.set(c.entry.id, forwardHits);
  }

  for (const c of candidates) {
    const forwardHits = forwardMap.get(c.entry.id)?.length ?? 0;
    const reverseHits = reverseIndex.get(c.entry.id)?.size ?? 0;
    const totalHits = forwardHits + reverseHits;
    if (totalHits > 0) {
      const boost = Math.min(totalHits * RELATION_BOOST_PER_HIT, RELATION_BOOST_CAP);
      c.score = Math.min(1, c.score + boost);
    }
  }
  return candidates;
}
```
